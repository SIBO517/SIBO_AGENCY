# Install webp tools (Linux/macOS)
brew install webp  # macOS
sudo apt-get install webp  # Ubuntu

# Convert hero image
cwebp -q 80 images/hero.jpg -o images/hero.webp

convert images/hero.jpg -resize 1200x images/hero-md.jpg
convert images/hero.jpg -resize 800x images/hero-sm.jpg
